# GitHub Setup Commands for vision-text-extractor
# Run these commands after creating the GitHub repository

# 1. Add the GitHub remote (replace YOUR_USERNAME with your actual GitHub username)
git remote add origin https://github.com/YOUR_USERNAME/vision-text-extractor.git

# 2. Verify the remote was added
git remote -v

# 3. Push to GitHub
git branch -M main
git push -u origin main

# 4. Verify the push was successful
git status

# Alternative: If you prefer SSH (if you have SSH keys set up)
# git remote add origin git@github.com:YOUR_USERNAME/vision-text-extractor.git
